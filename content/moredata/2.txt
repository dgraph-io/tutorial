# Define Types

type Person {
    name: string
    director.film: [Movie]
}

type Movie {
    name: string
    initial_release_date: string
    genre: [Genre]
    starring: [Performance]
}

type Genre {
    name: string
}

type Performance {
    performance.film: [Movie]
    performance.character: [Person]
    performance.actor: [Person]
}
